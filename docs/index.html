    draw_thick_line
    def draw_thick_line(Point a, Point b, number width, Color color) perp = perpendicular(norm(a - b)) * width gl.triangles([a - perp, a + perp, b + perp, b+perp, b-perp, a-perp] color) end

    approach
    def approach(number target, number step) : number state current = target if target > current current += min(step, target - current) elif target < current current -= min(step, current - target) end return current end

    button
    def button(Ref r, Point loc, string name) str = concat(name ' = ' r.to_string) text_sprite = text.render_text(ui_font_medium, str, #000) padding = 3.0 rect = [loc.x - padding, loc.y - padding, loc.x + text_sprite.width + padding, loc.y + text_sprite.height + padding] color = #ee3399 if mouse_in(rect) color = #ff44aa end draw_box(rect, color) text.draw_rendered_text(text_sprite, loc.x, loc.y) -- mouse wheel if mouse_wheel_up(rect) r.tweak(-1) elif mouse_wheel_down(rect) r.tweak(1) end -- click and drag state mouse_dragging = false state Point mouse_drag_previous_point = [0 0] if mouse_clicked(rect) mouse_dragging = true mouse_drag_previous_point = mouse elif not(mouse_pressed()) mouse_dragging = false end if mouse_dragging delta = (mouse - mouse_drag_previous_point) : Point mouse_drag_previous_point = mouse steps = round(delta.y) r.tweak(steps) end end
